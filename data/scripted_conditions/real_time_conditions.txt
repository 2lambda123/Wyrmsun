is_christmas_season = {
	real_month = december
	real_day = 16
}

is_easter_season = {
	#either March, or the first week of April
	or = {
		real_month = march
		and = {
			real_month = april
			not = {
				real_day = 8
			}
		}
	}
}

is_international_workers_day = {
	real_month = may
	real_day = 1
	not = {
		real_day = 2
	}
}
