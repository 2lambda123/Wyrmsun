can_spawn_gothic_minor_tribe = {
	or = {
		settlement_owner = {
			civilization = gothic
			not = {
				scripted_condition = has_high_medieval_technology
			}
		}
		any_neighbor_settlement_owner = {
			civilization = gothic
			not = {
				scripted_condition = has_high_medieval_technology
			}
		}
		any_settlement_neutral_building_owner = {
			civilization = gothic
			not = {
				scripted_condition = has_high_medieval_technology
			}
		}
		any_neighbor_settlement_neutral_building_owner = {
			civilization = gothic
			not = {
				scripted_condition = has_high_medieval_technology
			}
		}
	}
}
